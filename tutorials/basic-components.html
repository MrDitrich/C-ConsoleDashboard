Building Basic Dashboard Components

In this tutorial, you'll learn how to create reusable components for your console dashboard, including menus, tables, and progress bars.

Creating a Menu System

A menu system is essential for navigation in console applications. Here's a simple implementation:

public class Menu
{
    private List<string> options;
    private int selectedIndex;

    public Menu(List<string> options)
    {
        this.options = options;
        this.selectedIndex = 0;
    }

    public void Display()
    {
        Console.Clear();
        Console.WriteLine("Use arrow keys to navigate, Enter to select:");
        Console.WriteLine();
        
        for (int i = 0; i < options.Count; i++)
        {
            if (i == selectedIndex)
            {
                Console.ForegroundColor = ConsoleColor.Black;
                Console.BackgroundColor = ConsoleColor.White;
                Console.WriteLine($"► {options[i]}");
                Console.ResetColor();
            }
            else
            {
                Console.WriteLine($"  {options[i]}");
            }
        }
    }

    public int Run()
    {
        ConsoleKey key;
        do
        {
            Display();
            var keyInfo = Console.ReadKey(true);
            key = keyInfo.Key;

            if (key == ConsoleKey.UpArrow)
            {
                selectedIndex = Math.Max(0, selectedIndex - 1);
            }
            else if (key == ConsoleKey.DownArrow)
            {
                selectedIndex = Math.Min(options.Count - 1, selectedIndex + 1);
            }
        }
        while (key != ConsoleKey.Enter);

        return selectedIndex;
    }
}

Creating a Data Table

Display tabular data in a formatted way:

public class DataTable
{
    public static void Display(string[] headers, string[][] data)
    {
        // Calculate column widths
        int[] columnWidths = new int[headers.Length];
        for (int i = 0; i < headers.Length; i++)
        {
            columnWidths[i] = headers[i].Length;
            foreach (var row in data)
            {
                if (row[i].Length > columnWidths[i])
                    columnWidths[i] = row[i].Length;
            }
        }

        // Print header
        PrintRow(headers, columnWidths);
        PrintSeparator(columnWidths);

        // Print data
        foreach (var row in data)
        {
            PrintRow(row, columnWidths);
        }
    }

    private static void PrintRow(string[] row, int[] widths)
    {
        for (int i = 0; i < row.Length; i++)
        {
            Console.Write("| " + row[i].PadRight(widths[i]) + " ");
        }
        Console.WriteLine("|");
    }

    private static void PrintSeparator(int[] widths)
    {
        foreach (var width in widths)
        {
            Console.Write("+-" + new string('-', width) + "-");
        }
        Console.WriteLine("+");
    }
}

Creating a Progress Bar

Visual feedback for long-running operations:

public class ProgressBar
{
    public static void Show(int current, int total, string message = "")
    {
        int barWidth = 40;
        float percentage = (float)current / total;
        int filled = (int)(barWidth * percentage);

        Console.Write($"\r{message} [");
        Console.Write(new string('█', filled));
        Console.Write(new string('░', barWidth - filled));
        Console.Write($"] {percentage:P0}");
    }
}

int choice = menu.Run();
